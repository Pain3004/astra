pipeline {
    agent any
    stages {
        stage("Build") {
            steps {
                sh 'php --version'
                sh 'composer required mongodb/mongodb'
                sh 'composer update'
                sh 'cp .env.example .env'
                sh 'php artisan serve'
            }
        }
        stage("Unit test") {
            steps {
                sh 'php artisan test'
            }
        }
    }
}
